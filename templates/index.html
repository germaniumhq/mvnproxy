Create this as your .m2/settings.xml file. 

<settings>
  <mirrors>
    <mirror>
      <id>central</id>
      <name>central</name>
      <url>http://{{host}}:{{port}}/repo</url>
      <mirrorOf>*</mirrorOf>
    </mirror>
  </mirrors>
  <profiles>
    <profile>
      <activation>
        <activeByDefault>true</activeByDefault>
      </activation>
      <repositories>
        <repository>
          <id>central</id>
          <name>central</name>
          <url>http://{{host}}:{{port}}/repo</url>
          <releases>
            <enabled>true</enabled>
          </releases>
          <snapshots>
            <enabled>true</enabled>
          </snapshots>
        </repository>
      </repositories>
    </profile>
  </profiles>
</settings>
